/**
 * 函数：base64转化
 * @param  [string]  $data[参数]
 * @return [string]       [返回值]
 */
function base64_url_encode($data = []) {
    if (empty($data)) {
        return false;
    }
    return iconv('UTF-8', 'GB2312//IGNORE',  strtr(base64_encode($data), '+/=', '‐_,'));
}

base64解密:

/**
 * 函数：base64解密
 * @param  [string] $data [参数]
 * @return [string]       [返回值]
 */
function base64_url_decode($data) {
    return base64_decode(strtr($data, '‐_,', '+/='));
}

签名算法:

/**
 * 类：参数签名
 */
class Signature {

    // 盐：测试使用
    const SALT = 'testtest';

    /**
     * [sign 生成签名]
     * @param  [array] $args  [参数]
     * @return [string]       [返回值]
     */
    public static function sign($args) {
        if (!$args) {
            return '';
        }
        $salt = self::SALT;
        ksort($args);
        $signStr = '';
        foreach ($args as $k => $v) {
            if ($v || is_numeric($v)) {
                $signStr .= $k.$v;
            }
        }
        return strtoupper(md5(md5($signStr) . $salt));
    }
}

加密-解密算法:

/**
 * 类：加密-解密算法
 */
class HhrCrypt {

    // 密钥:测试使用
    const KEY = 'testtesttesttest';
    // 初始化向量:测试使用
    const IV  = 'testtesttesttest';

    /**
     * [encrypt 方法：加密]
     * @param  [string] $content [参数]
     * @return [string]          [返回值]
     */
    public static function encrypt($content)
    {
        $key     = self::KEY;
        $content = $content . "\0\0\0\0\0";
        $iv      = self::IV;
        $data    = openssl_encrypt($content, "aes-128-cbc", $key, OPENSSL_ZERO_PADDING, $iv);
        $error   = openssl_error_string();

        if($error) {
            return false;
        }
        return $data;
    }

    /**
     * [decrypt 方法：解密]
     * @param  [string] $encryData [参数]
     * @return [string]            [返回值]
     */
    public static function decrypt($encryData)
    {
        $key  = self::KEY;
        $iv   = self::IV;
        $data = openssl_decrypt($encryData, 'aes-128-cbc', $key, OPENSSL_ZERO_PADDING, $iv);
        $data = substr($data,0,11);
        return $data;
    }
}
